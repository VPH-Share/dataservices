# asio uri structure

# brood
/catalog
    /{schema}   -> PUT/DELETE to create/destroy schemas
        /sql
        /entity/(tables*)
        /schema
        /sparql
        /resources/(subjects*)
        /mapping
        /meta
    /events
/explore/(sql|sparql)
/insight/(spring-endpoints)
/whoami

# brood rewrites :
/{schema}/{role}/(.*)
  -> if {schema} is a deployed schema rewrite to
  /catalog/{schema}/(.*) and add header "Role={role}"
  -> else forward as is

# flock
/                                                 => snorql
/index.html                                       => snorql
/static/*                                         => js/css resources
/sparql                                           => query endpoint
/meta                                             => meta data
/insight/events                                   => event stream
