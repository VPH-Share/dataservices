# asio uri structure

# brood
/api                          => management endpoints
  /events                     => event stream
  /whoami                     => auth info
  /catalog/{schema}           => GET/PUT/DELETE to manage deployed schemas
/{schema}
  /sql                        => protocol
  /schema                     => sql schema
  /sparql                     => protocol
  /mapping                    => d2rq config
  /meta                       => vph metadata
  /explore/(sql|sparql)       => static content path - redirect to root
## handled by spring dispatcher
/static/{css, js, webjars, images}   => static content
/static/{sql, sparql, events}.html   => browser pages
/                             => welcome page

# brood default rewrites :
/{context}/{head}/{tail}
  # {context} is always omitted
  if {tail} startsWith '/explore'
      -> /static/{tail without '/explore'}
    if {head} is a deployed schema
      -> /{head}/{tail}

# brood VPH rewrites :
/{context}/{head}/{role}/{tail}
  # {context} is always omitted
  if {tail} startsWith '/explore'
    -> /static/{tail without '/explore'} and add login Role.NONE
  if {head} is a deployed schema
    -> /{head}/{tail} and add login '{role}'
